3. Group elements defined into families on the basis of their
sequence similarity:

        construction procedure with edge reevaluation (Fig. 5).

            void general_edge_redef(ELE_INFO_t *ele_info) {
                ELE_DATA_t *local_net=NULL;
                char *command;

                if (!ele_info->ele) ele_read_in(ele_info, 2);

                if (element has edges) {
                    build local edge network
                    cruise local edge network 
                    dissolve local edge network
                } else {
                    set element stat to 'y'
                    clean up element
                    release from graph
                }
            }
